# 535. Encode and Decode TinyURL

Two dictionaries:
* urlDB = {} # longURL -> code
* codeDB = {} # code -> longURL

Randomly generate 6 chars code:
* Use alphanumeric characters 0-9, a-z, A-Z as base.
* 62 in total. 62^6 different codes can be generated.

```python
class Codec:
    urlDB = {} # longURL -> code
    codeDB = {} # code -> longURL
    chars = string.digits + string.ascii_letters

    def encode(self, longUrl: str) -> str:
        """Encodes a URL to a shortened URL.
        """
        if longUrl in self.urlDB:
            return self.urlDB[longUrl]
        
        while True:
            short = "http://tinyurl.com/" + "".join(random.choice(self.chars) for _ in range(6))
            if short not in self.codeDB:
                self.codeDB[short] = longUrl
                break
        return short

    def decode(self, shortUrl: str) -> str:
        """Decodes a shortened URL to its original URL.
        """
        return(self.codeDB[shortUrl])
```
