* Data Structure:
stack = []. Use it to store numbers ONLY.

Consider all numbers in the stack should be added together at the end.
* If isdigit(): num = num * 10 + int(c)
* If +: stack.append(num)
* If -: stack.append(-num)
* If *: stack.append(stack_top * num)
* If /: stack.append(stack_top / num)



```python
class Solution:
    def calculate(self, s: str) -> int:
        mystack = []
        s = s.replace(" ", "")
        print(s)
        i = 0
        op = '+'
        cur_num = 0

        while i < len(s):
            #print(mystack, cur_num)

            if s[i].isdigit():
                cur_num = cur_num * 10 + int(s[i])

            if s[i] in '+-*/' or i == len(s) - 1:
                if op == '+':
                    mystack.append(cur_num)
                    cur_num = 0
                elif op == '-':
                    mystack.append(cur_num * -1)
                    cur_num = 0
                elif op == '*':
                    num1 = mystack.pop()
                    mystack.append(num1 * cur_num)
                    cur_num = 0
                else:
                    num1 = mystack.pop()
                    mystack.append(int(num1 / cur_num))
                    cur_num = 0

                op = s[i]
            
            i += 1
            #print(mystack, cur_num)
        
        result = 0
        while len(mystack):
            result += mystack.pop()
        
        return result
```
