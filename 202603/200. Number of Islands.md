# 200. Number of Islands
Iterate all nodes in the 2D list. 
From the first node of the island, visit all nodes that we can reach from the starting point, and mard it as visited.

* Time: O(n * m)
* Space: O(n * m)

``` python3
class Solution:

    def numIslands(self, grid: List[List[str]]) -> int:
        def visit(x, y):
            if x <0 or y < 0 or x == len(grid) or y == len(grid[0]):
                return
            if grid[x][y] == '0' or visited[x][y] == 1:
                return
            visited[x][y] = 1
            visit(x, y - 1)
            visit(x - 1, y)
            visit(x, y + 1)
            visit(x + 1, y)

        result = 0
        visited = [[0 for j in range(len(grid[0]))] for i in range(len(grid))]

        for i in range(len(grid)):
            for j in range(len(grid[i])):
                if grid[i][j] == '1':
                    if visited[i][j] == 1:
                        continue
                    else:
                        result += 1
                        visit(i, j)
        return result

```
